texto_pdf = ''

for i, pagina in enumerate(arquivo.pages):
    # Pega de cada página
    texto_pagina = pagina.extract_text()
    print(f'Página {i+1}')
    print(texto_pagina)
    
    # Exclui o começo e o fim do pdf que não irá utilizar
    texto_pagina = texto_pagina[295:]
    texto_pagina = texto_pagina[:-48]
    
    # Coloca tudo em uma variavel
    texto_pdf += f'{texto_pagina}\n'

referencia = []
qtd = []

data = []

# Conta quantas vezes o texto referencia aparece
quantidade = texto_pdf.lower().count(texto_referencia.lower())
posicao_final = 0
texto_analisar = texto_pdf
for i in range(quantidade):
    # Seleciona a parte do texto a partir da posi
    texto_analisar = texto_analisar[posicao_final:]
    
    # Pega em qual posição cada referencia está
    posicao_inicial = texto_analisar.find(texto_referencia)+15
    posicao_final = texto_analisar.find('TOTAL', posicao_inicial+1)

    posicao_data = posicao_inicial
    data_texto = texto_analisar[posicao_data-26:posicao_data-15]
    
    texto_final = texto_analisar[posicao_inicial:posicao_final]
    # Separa o texto 
    texto_final = texto_final.split('\n')
    texto_final = list(filter(None, texto_final))
    
    for texto in texto_final:
        texto = texto.split(' ')
        texto = list(filter(None, texto))
        
        n_venda.append(texto[0])
        referencia.append(texto[1])
        qtd.append(int(texto[2]))
        total.append(texto[3])
        produtos.append(texto[4:-1])
        tipo.append(texto[-1:])
        data.append(data_texto)
        
list_produtos = []
for produto in produtos:
    texto = ''
    for n in produto:
        texto += f' {n}'
    list_produtos.append(texto)
    
mes = []
dia = []
meses = ['Jan', 'Fev', 'Mar', 'Abr', 'Mai', 'Jun', 'Jul', 'Ago', 'Set', 'Out', 'Nov', 'Dez']
for i, d in enumerate(data):
    data[i] = d.replace('\n', '')
    data[i] = datetime.strptime(data[i], "%d/%m/%Y")
    data[i] = data[i].date()
    dia.append(int(data[i].day))
    mes.append(meses[int(data[i].month)-1])
    

for i, t in enumerate(tipo):
    tipo[i] = t[0]

    total[i] = t.replace(',', '.')
    total[i] = float(total[i])
